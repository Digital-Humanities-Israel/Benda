בְּיוֹם קַיִץ, יוֹם חֹם

 

בְּיוֹם קַיִץ, יוֹם חֹם, עֵת הַשֶּׁמֶשׁ מִמְּרוֹם

הָרָקִיעַ תְּלַהֵט כַּתַּנּוּר הַיּוֹם,

עֵת יְבַקֵּשׁ הַלֵּב פִּנַּת שֶׁקֶט לַחֲלוֹם –

בֹּא אֵלַי, בֹּא אֵלַי, רֵעַ עָיֵף!

 

לִי-יֶשׁ-גָּן, וּבַגָּן תַּחַת אוֹג כֶּבֶד צֵל,

הַרְחֵק הַרְחֵק מֵעִיר וּמִמְּתִים, נֶחְבָּא תֵל,

כֻּלּוֹ עָטוּף יְרַקְרַק, כֻּלּוֹ אֹמֵר סוֹד אֵל –

שָׁם נֵחָבֵא, נָנוּחַ, אָח נָעִים!

 

וּבְהִתְעַנֵּג וּבְרֹךְ שָׁם בַּמַּחֲבֵא הַלָּז

נַמְתִּיק יַחְדָּו בִּדְמִי צָהֳרַיִם הָרָז,

שֶׁיּוֹפִיעַ לִשְׁנֵינוּ מִקֶּרֶן הַפָּז,

הַבֹּקַעַת אֶת-שִׁפְעַת הַצְּלָלִים.

 

וּבְלֵיל חֹרֶף, לֵיל קֹר, עֵת מַחֲשַׁכִּים וּשְׁחוֹר

יְשׁוּפוּךָ בַּחוּץ, הוֹלֵךְ סוֹבֵב הַקֹּר,

וּבִבְשָׂרְךָ כִּי-יִתְקַע מַאַכְלוֹתָיו הַכְּפוֹר –

בֹּא אֵלַי, בֹּא אֵלַי, בְּרוּךְ אֲדֹנָי!

 

בֵּיתִי קָטָן וָדַל, בְּלִי מַכְלוּלִים וּפְאֵר,

אַךְ הוּא חָם, מָלֵא אוֹר וּפָתוּחַ לַגֵּר,

עַל-הָאָח בֹּעֵר אֵשׁ, עַל-הַשֻּׁלְחָן הַנֵּר –

אֶצְלִי שֵׁב וְהִתְחַמֵּם, אָח אֹבֵד!

 

וּבְהִשָּׁמַע מִילֵל סוּפַת לֵיל קוֹל כָּאוֹב,

זָכֹר נִזְכֹּר עֱנוּת רָשׁ גֹּוֵע בָּרְחוֹב,

וּלְחַצְתִּיךָ אֶל-לֵב, רֵעִי, אָחִי הַטּוֹב –

וּרְסִיס נֶאֱמָן אוֹרִידָה עָלֶיךָ.

 

אַךְ בְּבֹא תוֹר הַסְּתָו, בִּימֵי סַגְרִיר וָעָב;

עָמוּם שׁוֹמֵם הַיְקוּם, רֶפֶשׁ הוֹלֵךְ וָרָב,

דֶּלֶף טֹרֵד עַל-גָּג, עָשׁ בְּקֶרֶב לֵבָב –

בִּי, עָזְבֵנִי לְנַפְשִׁי, אָח חַנּוּן!

 

בַּשִּׁמָּמוֹן הַזֶּה אֶחְפֹּץ בָּדָד הֱיוֹת;

וּבְהִתְעַטֵּף הַלֵּב וּבִהְיוֹתוֹ לִמְסוֹס

אַל-תִּרְאֵנִי עֵין זָר, זָר לֹא-יָבִין אֶת-זֹאת –

וַעֲרִירִי אַחֲרִישָׁה בִיגוֹנִי.

 

חשון, תרנ"ז.


לתוכן הענינים

לדף הראשי של פרויקט בן-יהודה
